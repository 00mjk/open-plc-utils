#ifndef SET32BITMAP_SOURCE
#define SET32BITMAP_SOURCE

#include "../tools/flags.h"
#include "../tools/endian.h"

void set32bitmap (uint32_t * map, unsigned bit) 

{
	map [bit / _bits (* map)] |= HTOLE32 (1 << (bit % _bits (* map)));
	return;
}

#endif
