#ifndef UPGRADEDEVICE1_SOURCE
#define UPGRADEDEVICE1_SOURCE

#include "../plc/plc.h"

int UpgradeDevice1 (struct plc * plc) 

{
	if (WriteNVM (plc)) 
	{
		return (-1);
	}
	if (WritePIB (plc)) 
	{
		return (-1);
	}
	if (FlashDevice1 (plc)) 
	{
		return (-1);
	}
	return (0);
}
