char const * MMEName (uint16_t MMTYPE) 

{
	size_t lower = 0;
	size_t upper = SIZEOF (mme_names);
	MMTYPE &= MMTYPE_MASK;
	while (lower < upper) 
	{
		size_t index = (lower + upper) >> 1;
		signed order = MMTYPE - mme_names [index].type;
		if (order < 0) 
		{
			upper = index - 0;
			continue;
		}
		if (order > 0) 
		{
			lower = index + 1;
			continue;
		}
		return (mme_names [index].name);
	}
	return ("UNKNOWN_MESSAGE_TYPE");
}
